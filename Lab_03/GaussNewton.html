# We define ellipse function.
def func(p, x):
    a, b, c = p
    return x[:, 0] ** 2 / a**2 + x[:, 1] ** 2 / b**2 + x[:, 2] ** 2 / c**2


# Then define the residual function.
def residual(p, x):
    res = func(p, x) - 1
    return res

def res(param, data):
    """
    Args:
        param: the parameters you want to optimize
        data: the data you need to fit
    Returns:
        res: the residual of the points with objective function
    """
    a , b , c = param
    res = np.zeros(data.shape[0])
    for i in range(data.shape[0]):
        res_val = data[i, 0] ** 2 / a**2 + data[i, 1] ** 2 / b**2 + data[i, 2] ** 2 / c**2 - 1
        res[i] = res_val
    return res


def jacobi(param, data):
    """
    Args:
        param: the parameters you want to optimize
        data: the data you need to fit
    Returns:
        d: the jacobi of the objective function with these data points and parameters
    """
    a , b , c = param
    d = np.zeros((data.shape[0], 3))
    for i in range(data.shape[0]):
        d[i, 0] = -2 * data[i, 0] ** 2 / a**3
        d[i, 1] = -2 * data[i, 1] ** 2 / b**3
        d[i, 2] = -2 * data[i, 2] ** 2 / c**3
    return d


def solve_delta(params, data):
    """
    Args:
        param: the parameters you want to optimize
        data: the data you need to fit
    Returns:
        delta_p: the optimal direction of the parameters
    """
    res_vector = res(params, data)
    J = jacobi(params, data)
    JT = np.transpose(J)
    JTJ_inverse = np.linalg.inv(np.dot(JT, J))
    delta_p = - np.dot(np.dot(JTJ_inverse, JT), res_vector)
    return delta_p
